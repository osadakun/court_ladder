```typescript
export function canRollback(ctx: {
  matchState: string;
  matchType: string;
  isLatestFinishedOnCourt: boolean;
  winnerInQueue: boolean;
  loserInQueue: boolean;
  winnerInNewMatch: boolean;
  loserInNewMatch: boolean;
  winnerFinishedNewMatch: boolean;
  loserFinishedNewMatch: boolean;
}): { allowed: boolean; errorCode: string | null } {
  if (ctx.matchState !== "finished") {
    return { allowed: false, errorCode: "MATCH_NOT_FINISHED" };
  }
  if (ctx.matchType !== "regular") {
    return { allowed: false, errorCode: "AUTO_ROLLBACK_NOT_ALLOWED" };
  }
  if (!ctx.isLatestFinishedOnCourt) {
    return { allowed: false, errorCode: "AUTO_ROLLBACK_NOT_ALLOWED" };
  }
  if (ctx.winnerFinishedNewMatch || ctx.loserFinishedNewMatch) {
    return { allowed: false, errorCode: "AUTO_ROLLBACK_NOT_ALLOWED" };
  }
  const winnerReachable = ctx.winnerInQueue || ctx.winnerInNewMatch;
  const loserReachable = ctx.loserInQueue || ctx.loserInNewMatch;
  if (!winnerReachable || !loserReachable) {
    return { allowed: false, errorCode: "AUTO_ROLLBACK_NOT_ALLOWED" };
  }
  return { allowed: true, errorCode: null };
}
```

```typescript
export function pickMatchEntries(queue: QueueEntry[]): {
  entryA: string;
  entryB: string;
  entryAOriginalQueuePosition: number;
  entryBOriginalQueuePosition: number;
  remainingQueue: QueueEntry[];
} {
  const sorted = [...queue].sort((a, b) => a.queuePosition - b.queuePosition);
  return {
    entryA: sorted[0].entryId,
    entryB: sorted[1].entryId,
    entryAOriginalQueuePosition: sorted[0].queuePosition,
    entryBOriginalQueuePosition: sorted[1].queuePosition,
    remainingQueue: sorted.slice(2).map((q, i) => ({
      entryId: q.entryId,
      queuePosition: i + 1,
    })),
  };
}
```

- [ ] **Step B3: restoreToOriginalPosition を実装**

アルゴリズム: 既存キューのソート済み配列に、復元エントリーを元のposition位置に挿入し、全体を1から再採番する。

```typescript
/**
 * FR-10 v2.7 ロールバック時に元の位置にエントリーを復元する
 *
 * 復元エントリーを元の queue_position の位置に挿入し、
 * 既存アイテムとマージして1から再採番する。
 *
 * 元位置が現在の待機列サイズを超える場合は末尾に追加する。
 */
export function restoreToOriginalPosition(
  queue: QueueEntry[],
  entryAId: string,
  entryAOrigPos: number,
  entryBId: string,
  entryBOrigPos: number,
): QueueEntry[] {
  const sorted = [...queue].sort((a, b) => a.queuePosition - b.queuePosition);

  // 復元エントリーを位置順にソート
  const toRestore = [
    { entryId: entryAId, origPos: entryAOrigPos },
    { entryId: entryBId, origPos: entryBOrigPos },
  ].sort((a, b) => a.origPos - b.origPos);

  // 既存キューに復元エントリーを挿入
  // origPos は「元のキュー全体での位置」を表す
  // 既存キューの i 番目の前に、origPos <= i+1 の復元エントリーを挿入
  const result: QueueEntry[] = [];
  let restoreIdx = 0;
  let existIdx = 0;
  let currentOrigPos = 1; // 元のキュー全体での現在位置

  while (restoreIdx < toRestore.length || existIdx < sorted.length) {
    // 復元エントリーの挿入タイミング: 元のpositionが現在位置と一致
    if (restoreIdx < toRestore.length && toRestore[restoreIdx].origPos <= currentOrigPos) {
      result.push({ entryId: toRestore[restoreIdx].entryId, queuePosition: result.length + 1 });
      restoreIdx++;
      currentOrigPos++;
      continue;
    }

    if (existIdx < sorted.length) {
      result.push({ entryId: sorted[existIdx].entryId, queuePosition: result.length + 1 });
      existIdx++;
      currentOrigPos++;
    } else {
      // 既存が尽きたら残りの復元エントリーを末尾に追加
      result.push({ entryId: toRestore[restoreIdx].entryId, queuePosition: result.length + 1 });
      restoreIdx++;
      currentOrigPos++;
    }
  }

  return result;
}
```

2. その試合を `cancelled` (`cancel_reason = 'rollback'`) に更新

4. 当該 `in_progress` 試合の **両エントリー** を待機列先頭に戻す

```typescript
async function cancelInProgressMatchForRollback(
  db: SupabaseClient,
  tournamentId: string,
  inProgressMatch: {
    match_id: string;
    court_no: number;
    entry_a_id: string;
    entry_b_id: string;
    version: number;
  },
): Promise<void> {
  // 試合をキャンセル
  await db.from("matches").update({
    state: "cancelled",
    cancelled_at: new Date().toISOString(),
    cancel_reason: "rollback",
    version: inProgressMatch.version + 1,
  }).eq("match_id", inProgressMatch.match_id);

  // current_match_id をクリア
  await db.from("courts").update({ current_match_id: null })
    .eq("tournament_id", tournamentId)
    .eq("court_no", inProgressMatch.court_no);

  // 両エントリーを待機列先頭に戻す（一括delete+insert方式）
  // 既存キューを取得
  const { data: queue } = await db.from("queue_items")
    .select("queue_item_id, entry_id, queue_position, enqueue_reason, source_match_id")
    .eq("tournament_id", tournamentId)
    .eq("court_no", inProgressMatch.court_no)
    .order("queue_position");

  // 一括delete
  await db.from("queue_items")
    .delete()
    .eq("tournament_id", tournamentId)
    .eq("court_no", inProgressMatch.court_no);

  // entry_a → entry_b の順で先頭へ戻し、既存を後方にシフト
  await db.from("queue_items").insert([
    {
      tournament_id: tournamentId,
      court_no: inProgressMatch.court_no,
      entry_id: inProgressMatch.entry_a_id,
      queue_position: 1,
      enqueue_reason: "rollback",
      source_match_id: inProgressMatch.match_id,
    },
    {
      tournament_id: tournamentId,
      court_no: inProgressMatch.court_no,
      entry_id: inProgressMatch.entry_b_id,
      queue_position: 2,
      enqueue_reason: "rollback",
      source_match_id: inProgressMatch.match_id,
    },
  ]);
  for (let i = 0; i < (queue || []).length; i++) {
    await db.from("queue_items").insert({
      tournament_id: tournamentId,
      court_no: inProgressMatch.court_no,
      entry_id: queue![i].entry_id,
      queue_position: i + 3,
      enqueue_reason: queue![i].enqueue_reason || "rollback",
      source_match_id: queue![i].source_match_id,
    });
  }

  // movement_log 記録（両エントリー分）
  await db.from("movement_logs").insert([
    {
      tournament_id: tournamentId,
      match_id: inProgressMatch.match_id,
      entry_id: inProgressMatch.entry_a_id,
      from_court_no: null,
      to_court_no: inProgressMatch.court_no,
      movement_reason: "rollback",
    },
    {
      tournament_id: tournamentId,
      match_id: inProgressMatch.match_id,
      entry_id: inProgressMatch.entry_b_id,
      from_court_no: null,
      to_court_no: inProgressMatch.court_no,
      movement_reason: "rollback",
    },
  ]);
}
```

## 検証方法

1. `$HOME/.deno/bin/deno test tests/core/` で全コアロジックテスト GREEN
2. ローカル Supabase でマイグレーション適用（`supabase db reset`）
3. リクエスト試合: 作成 → 結果入力 → 履歴に「リクエスト」表示 → 削除
4. ロールバック: 結果確定 → ロールバック → 元の位置(1,2)に復元 → 同じ対戦が再生成
5. ロールバック: 元の位置(3,4)または非連続位置に復元 → 既存待機列が後方シフト
6. ロールバック: 結果確定 → 移動先で試合中 → ロールバック → 移動先試合もキャンセル → 元の位置に復元
7. ロールバック: 2件前の試合 → 409 AUTO_ROLLBACK_NOT_ALLOWED
8. 結果再入力: finished 試合のスコア変更 → 履歴更新 → 移動に影響なし
9. D&D: 待機列内並び替え + コート間移動 + current_match 中エントリー移動
10. 公開画面: public_token付きURLで正しく表示、リクエスト試合が非表示、Realtime で更新反映
11. `npm run build` でフロントエンドビルド成功
